#!/bin/bash

echo "Step 0: Check Docker Installation"
docker version
docker info

echo
echo "Step 1: Pull Python Image"
docker pull python:3.9.18-slim-bullseye

echo
echo "Step 2: List Docker Images"
docker images
docker images -q
docker images -a
docker image ls

echo
echo "Step 3: Run a Container (detached mode to show in docker ps)"
docker run -d --name mypython python:3.9.18-slim-bullseye tail -f /dev/null

echo
echo "Step 4: Check Running Containers"
docker ps

echo
echo "Step 5: Check All Containers"
docker ps -a

echo
echo "Step 6: Inspect Image and Container"
docker inspect python:3.9.18-slim-bullseye
docker inspect mypython

echo
echo "Step 7: Stop the Container"
docker stop mypython

echo
echo "Step 8: Verify Container Stopped"
docker ps -a

echo
echo "Step 9: Remove the Container"
docker rm mypython

echo
echo "Step 10: Remove the Image"
docker rmi python:3.9.18-slim-bullseye

echo
echo "Docker Practical Exam Completed Successfully!"










# Create a new project folder
mkdir GitDemo
cd GitDemo

# Initialize Git repository
git init

# Add a file
echo "Hello from Shreyas" > demo.txt
git status

# Add file to staging area
git add demo.txt

# Commit the changes
git commit -m "Initial commit by Shreyas"

# Connect local repo with remote repository
git remote add origin https://github.com/shreyas/GitDemo.git

# Push files to remote repository
git branch -M main
git push -u origin main










# Create and initialize repository
mkdir GitBranchDemo
cd GitBranchDemo
git init

# Add and commit files
echo "Main branch file by Shreyas" > main.txt
git add .
git commit -m "Main branch initial commit by Shreyas"

# Create a new branch
git branch feature

# Switch to the new branch
git checkout feature

# Add files in feature branch
echo "Feature branch changes by Shreyas" > feature.txt
git add .
git commit -m "Added feature file by Shreyas"

# Switch back to main branch
git checkout main

# Merge branch
git merge feature







# Clone repository to local system
git clone https://github.com/shreyas/RemoteGitDemo.git
cd RemoteGitDemo

# Add new files
echo "This is Shreyas's remote demo" > remote.txt
git add .
git commit -m "Added remote demo file by Shreyas"

# Push changes to remote
git push origin main

# Pull latest updates
git pull origin main
